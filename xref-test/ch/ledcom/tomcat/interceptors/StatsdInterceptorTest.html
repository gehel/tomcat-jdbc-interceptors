29  import org.apache.tomcat.jdbc.pool.DataSource;

31  import org.hsqldb.jdbc.JDBCDriver;
32  import org.junit.After;
33  import org.junit.Before;
34  import org.junit.Test;

36  public class StatsdInterceptorTest {
37  
38      private static final String STATSD_PREFIX = "jdbc.pool.";
39      private static final int STATSD_PORT = 6545;
40      private DataSource ds;
41      private Thread statsdThread;
42      private List<String> receivedMessages = new CopyOnWriteArrayList<String>();
43      private List<IOException> statsdExceptions = new CopyOnWriteArrayList<IOException>();
44  
45      private final Object packetReceived = new Object();
46  
47      @Before
48      public void createDataSource() {
49          PoolProperties poolProperties = new PoolProperties();
50          poolProperties.setDriverClassName(JDBCDriver.class.getName());
51          poolProperties.setUrl("jdbc:hsqldb:mem:aname");
52          poolProperties.setUsername("sa");
53          poolProperties.setPassword("");
54          poolProperties.setJdbcInterceptors(StatsdInterceptor.class.getName()
55                  + "(hostname=localhost,port=" + STATSD_PORT
56                  + ",sampleRate=1.0,prefix=" + STATSD_PREFIX + ")");
57          ds = new DataSource();
58          ds.setPoolProperties(poolProperties);
59      }
60  
61      @Before
62      public void startMockStatsd() {
63          statsdThread = new Thread(new Runnable() {
64  
65              @Override
66              public void run() {
67                  try {
68                      DatagramSocket serverSocket = new DatagramSocket(
69                              STATSD_PORT);
70                      serverSocket.setSoTimeout(500);
71                      while (!Thread.currentThread().isInterrupted()) {
72                          byte[] receiveData = new byte[1024];
73                          DatagramPacket receivePacket = new DatagramPacket(
74                                  receiveData, receiveData.length);
75                          try {
76                              serverSocket.receive(receivePacket);
77                          } catch (SocketTimeoutException expected) {
78                              continue;
79                          }
80                          String message = new String(receivePacket.getData(), 0,
81                                  receivePacket.getLength());
82                          System.out.println(message);
83                          receivedMessages.add(message);
84                          synchronized (packetReceived) {
85                              packetReceived.notifyAll();
86                          }
87                      }
88                  } catch (IOException ioe) {
89                      statsdExceptions.add(ioe);
90                  }
91              }
92          });
93          statsdThread.start();
94      }
95  
96      @Test
97      public void interceptorSendsPacketsToStatsd() throws SQLException,
98              InterruptedException, IOException {
99          Connection conn = ds.getConnection();
100         Statement stmt = conn.createStatement();
101         stmt.execute("Create table toto (a Integer)");
102         stmt.close();
103         conn.close();
104 
105         // give time to the packet to be received
106         waitForFirstPacket();
107 
108         if (statsdExceptions.size() > 0) {
109             throw statsdExceptions.get(0);
110         }
111 
112         assertTrue("Exception where thrown by mock Statsd",
113                 statsdExceptions.size() == 0);
114         assertTrue("No packet was received by mock Statsd.",
115                 receivedMessages.size() > 0);
116         assertEquals("jdbc.pool.createStatement.count:1|c",
117                 receivedMessages.get(0));
118     }
119 
120     @After
121     public void closeDataSource() {
122         ds.close();
123     }
124 
125     @After
126     public void stopMockStatsd() throws InterruptedException {
127         statsdThread.interrupt();
128         statsdThread.join();
129     }
130 
131     private void waitForFirstPacket() throws InterruptedException {
132         if (receivedMessages.size() == 0) {
133             synchronized (packetReceived) {
134                 packetReceived.wait(1000);
135             }
136         }
137     }
138 }
