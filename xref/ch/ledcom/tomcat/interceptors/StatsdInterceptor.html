14  package ch.ledcom.tomcat.interceptors;

16  import java.io.IOException;
17  import java.lang.reflect.Method;
18  import java.net.InetSocketAddress;
19  import java.nio.ByteBuffer;
20  import java.nio.channels.DatagramChannel;
21  import java.util.Locale;
22  import java.util.Map;
23  import java.util.Random;
24  import java.util.logging.Level;
25  import java.util.logging.Logger;

27  import org.apache.tomcat.jdbc.pool.ConnectionPool;
28  import org.apache.tomcat.jdbc.pool.JdbcInterceptor;
29  import org.apache.tomcat.jdbc.pool.PoolProperties;
30  import org.apache.tomcat.jdbc.pool.PoolProperties.InterceptorProperty;
31  import org.apache.tomcat.jdbc.pool.PooledConnection;

33  public class StatsdInterceptor extends JdbcInterceptor {
34  
35      private static final int BUFFER_SIZE = 1500;
36  
37      private final ByteBuffer sendBuffer = ByteBuffer.allocate(BUFFER_SIZE);
38  
39      private static final Random RNG = new Random();
40  
41      private static final Logger log = Logger.getLogger(StatsdInterceptor.class
42              .getName());
43  
44      private InetSocketAddress address;
45      private DatagramChannel channel;
46      private double sampleRate;
47      private String prefix;
48  
49      /**
50       * This interceptor has no state to be reset, so this method does nothing.
51       * 
52       * @see JdbcInterceptor#reset(ConnectionPool, PooledConnection)
53       */
54      @Override
55      public void reset(final ConnectionPool parent, final PooledConnection conn) {
56          // do nothing on reset, this interceptor has no state except
57          // configuration
58      }
59  
60      /**
61       * Configure the interceptor.
62       * 
63       * The following options are all required :
64       * <ul>
65       * <li>hostname: the hostname of the Statsd server</li>
66       * <li>port: the port of the Statsd server</li>
67       * <li>sampleRate: the fraction of connections that will be measured</li>
68       * <li>prefix: this prefix will be added to the keys published to Statsd</li>
69       * </ul>
70       */
71      @Override
72      public synchronized void setProperties(
73              final Map<String, PoolProperties.InterceptorProperty> properties) {
74          super.setProperties(properties);
75          InterceptorProperty hostnameProp = properties.get("hostname");
76          if (hostnameProp == null) {
77              throw new IllegalArgumentException(
78                      "property \"hostname\" has not been set");
79          }
80          InterceptorProperty portProp = properties.get("port");
81          if (portProp == null) {
82              throw new IllegalArgumentException(
83                      "property \"port\" has not been set");
84          }
85          InterceptorProperty sampleRateProp = properties.get("sampleRate");
86          if (sampleRateProp == null) {
87              throw new IllegalArgumentException(
88                      "property \"sampleRate\" has not been set");
89          }
90          InterceptorProperty prefixProp = properties.get("prefix");
91          if (prefixProp == null) {
92              throw new IllegalArgumentException(
93                      "property \"prefix\" has not been set");
94          }
95  
96          String hostname = hostnameProp.getValue();
97          int port = portProp.getValueAsInt(0);
98          sampleRate = sampleRateProp.getValueAsDouble(1.0);
99          prefix = prefixProp.getValue();
100         address = new InetSocketAddress(hostname, port);
101         try {
102             channel = DatagramChannel.open();
103         } catch (IOException ioe) {
104             throw new RuntimeException(ioe);
105         }
106     }
107 
108     /**
109      * Gets invoked each time an operation on {@link java.sql.Connection} is
110      * invoked.
111      * 
112      * The timing and count of each method calls to {@link java.sql.Connection}
113      * are sent to Statsd.
114      */
115     @Override
116     public Object invoke(final Object proxy, final Method method,
117             final Object[] args) throws Throwable {
118         String methodName = method.getName();
119         long start = System.nanoTime();
120         Object result = super.invoke(proxy, method, args);
121 
122         increment(methodName + ".count");
123         timing(methodName + ".timing", System.nanoTime() - start);
124         return result;
125     }
126 
127     private boolean increment(final String key) {
128         return send(sampleRate,
129                 String.format(Locale.ENGLISH, "%s:1|c", prefix + key));
130     }
131 
132     private boolean timing(final String key, final long value) {
133         return send(sampleRate,
134                 String.format(Locale.ENGLISH, "%s:%d|ms", prefix + key, value));
135     }
136 
137     private boolean send(final double sampleRate, String stat) {
138 
139         boolean retval = false; // didn't send anything
140         if ((sampleRate < 1.0) && (RNG.nextDouble() <= sampleRate)) {
141             stat = String.format(Locale.ENGLISH, "%s|@%f", stat, sampleRate);
142         }
143         if (doSend(stat)) {
144             retval = true;
145         }
146 
147         return retval;
148     }
149 
150     private synchronized boolean doSend(final String stat) {
151         try {
152             final byte[] data = stat.getBytes("utf-8");
153 
154             // If we're going to go past the threshold of the buffer then flush.
155             // the +1 is for the potential '\n' in multi_metrics below
156             if (sendBuffer.remaining() < (data.length + 1)) {
157                 flush();
158             }
159 
160             if (sendBuffer.position() > 0) { // multiple metrics are separated
161                                              // by '\n'
162                 sendBuffer.put((byte) '\n');
163             }
164 
165             sendBuffer.put(data); // append the data
166 
167             flush();
168 
169             return true;
170 
171         } catch (IOException e) {
172             log.log(Level.WARNING, String.format(
173                     "Could not send stat %s to host %s:%d",
174                     sendBuffer.toString(), address.getHostName(),
175                     address.getPort()), e);
176             return false;
177         }
178     }
179 
180     private synchronized boolean flush() {
181         try {
182             final int sizeOfBuffer = sendBuffer.position();
183 
184             if (sizeOfBuffer <= 0) {
185                 return false;
186             } // empty buffer
187 
188             // send and reset the buffer
189             sendBuffer.flip();
190             final int nbSentBytes = channel.send(sendBuffer, address);
191             sendBuffer.limit(sendBuffer.capacity());
192             sendBuffer.rewind();
193 
194             if (sizeOfBuffer == nbSentBytes) {
195                 return true;
196             } else {
197                 log.log(Level.WARNING,
198                         String.format(
199                                 "Could not send entirely stat %s to host %s:%d. Only sent %d bytes out of %d bytes",
200                                 sendBuffer.toString(), address.getHostName(),
201                                 address.getPort(), nbSentBytes, sizeOfBuffer));
202                 return false;
203             }
204 
205         } catch (IOException e) {
206             log.log(Level.WARNING, String.format(
207                     "Could not send stat %s to host %s:%d",
208                     sendBuffer.toString(), address.getHostName(),
209                     address.getPort()), e);
210             return false;
211         }
212     }
