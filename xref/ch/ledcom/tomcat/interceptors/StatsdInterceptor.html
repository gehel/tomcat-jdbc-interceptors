32  /**
33   * {@link JdbcInterceptor} reporting timing of certain method calls to a Statsd
34   * server.
35   *
36   * @author gehel
37   */
38  public class StatsdInterceptor extends JdbcInterceptor {
39  
40      /** Only report metrics for those methods. */
41      private static final Set<String> METHODS_TO_REPORT = new HashSet<String>(
42              Arrays.asList("commit", "createStatement", "getMetadata",
43                      "prepareCall", "prepareStatement", "rollback"));
44      /** {@link Metrics} used for reporting. */
45      private Metrics metrics;
46  
47      /** Factory to create {@link Statement} proxy. */
48      private ProxyFactory proxyFactory;
49  
50      /**
51       * This interceptor has no state to be reset, so this method does nothing.
52       *
53       * @see JdbcInterceptor#reset(ConnectionPool, PooledConnection)
54       * @param parent
55       *            the connection pool owning the connection
56       * @param conn
57       *            the pooled connection
58       */
59      @Override
60      public void reset(final ConnectionPool parent,
61              final PooledConnection conn) {
62          // do nothing on reset, this interceptor has no state except
63          // configuration
64      }
65  
66      /**
67       * Configure the interceptor.
68       *
69       * The following options are all required :
70       * <ul>
71       * <li>hostname: the hostname of the Statsd server</li>
72       * <li>port: the port of the Statsd server</li>
73       * <li>sampleRate: the fraction of connections that will be measured</li>
74       * <li>prefix: this prefix will be added to the keys published to
75       * Statsd</li>
76       * </ul>
77       *
78       * @param properties configuration for the interceptor
79       */
80      @Override
81      public final void setProperties(
82              final Map<String, PoolProperties.InterceptorProperty> properties) {
83          super.setProperties(properties);
84          InterceptorProperty hostnameProp = properties.get("hostname");
85          if (hostnameProp == null) {
86              throw new IllegalArgumentException(
87                      "property \"hostname\" has not been set");
88          }
89          InterceptorProperty portProp = properties.get("port");
90          if (portProp == null) {
91              throw new IllegalArgumentException(
92                      "property \"port\" has not been set");
93          }
94          InterceptorProperty sampleRateProp = properties.get("sampleRate");
95          if (sampleRateProp == null) {
96              throw new IllegalArgumentException(
97                      "property \"sampleRate\" has not been set");
98          }
99          InterceptorProperty prefixProp = properties.get("prefix");
100         if (prefixProp == null) {
101             throw new IllegalArgumentException(
102                     "property \"prefix\" has not been set");
103         }
104 
105         double sampleRate = sampleRateProp.getValueAsDouble(1.0);
106         metrics = new Metrics(hostnameProp.getValue(),
107                 portProp.getValueAsInt(0), prefixProp.getValue(), sampleRate);
108         proxyFactory = new ProxyFactory(metrics);
109     }
110 
111     /**
112      * Gets invoked each time an operation on {@link java.sql.Connection} is
113      * invoked.
114      *
115      * The timing and count of each method calls to {@link java.sql.Connection}
116      * are sent to Statsd.
117      *
118      * {@inheritDoc}
119      */
120     @Override
121     public final Object invoke(final Object proxy, final Method method,
122             final Object[] args) throws Throwable {
123         String methodName = method.getName();
124         boolean sample = metrics.sample();
125         boolean shouldReport = shouldReport(methodName);
126         long start = 0;
127         if (sample && shouldReport) {
128             start = System.nanoTime();
129         }
130         try {
131             Object o = super.invoke(proxy, method, args);
132             if (!sample) {
133                 // if this call is not sampled, no need to proxy the statements
134             } else if ("createStatement".equals(methodName)) {
135                 o = proxyFactory.statementProxy((Statement) o);
136             } else if ("prepareStatement".equals(method.getName())) {
137                 o = proxyFactory.preparedStatementProxy((PreparedStatement) o);
138             } else if ("prepareCall".equals(method.getName())) {
139                 o = proxyFactory.callableStatementProxy((CallableStatement) o);
140             }
141             return o;
142         } catch (InvocationTargetException e) {
143             throw e.getCause();
144         } finally {
145             if (sample && shouldReport) {
146                 metrics.timing(".connection." + methodName + ".timing",
147                         System.nanoTime() - start);
148             }
149         }
150     }
151 
152     /**
153      * Decide if we should report metrics for a specific method or not.
154      *
155      * @param methodName
156      *            name of the method
157      * @return <code>true</code> if we should report metrics
158      */
159     private boolean shouldReport(final String methodName) {
160         return METHODS_TO_REPORT.contains(methodName);
161     }
