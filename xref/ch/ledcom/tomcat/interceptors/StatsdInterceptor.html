14  package ch.ledcom.tomcat.interceptors;
15  
16  import java.lang.reflect.InvocationTargetException;
17  import java.lang.reflect.Method;
18  import java.sql.CallableStatement;
19  import java.sql.PreparedStatement;
20  import java.sql.Statement;
21  import java.util.Map;
22  
23  import org.apache.tomcat.jdbc.pool.ConnectionPool;
24  import org.apache.tomcat.jdbc.pool.JdbcInterceptor;
25  import org.apache.tomcat.jdbc.pool.PoolProperties;
26  import org.apache.tomcat.jdbc.pool.PoolProperties.InterceptorProperty;
27  import org.apache.tomcat.jdbc.pool.PooledConnection;
28  
29  public class StatsdInterceptor extends JdbcInterceptor {
30  
31      private Metrics metrics;
32  
33      private ProxyFactory proxyFactory;
34  
35      /**
36       * This interceptor has no state to be reset, so this method does nothing.
37       * 
38       * @see JdbcInterceptor#reset(ConnectionPool, PooledConnection)
39       * @param parent
40       *            the connection pool owning the connection
41       * @param con
42       *            the pooled connection
43       */
44      @Override
45      public void reset(final ConnectionPool parent, final PooledConnection conn) {
46          // do nothing on reset, this interceptor has no state except
47          // configuration
48      }
49  
50      /**
51       * Configure the interceptor.
52       * 
53       * The following options are all required :
54       * <ul>
55       * <li>hostname: the hostname of the Statsd server</li>
56       * <li>port: the port of the Statsd server</li>
57       * <li>sampleRate: the fraction of connections that will be measured</li>
58       * <li>prefix: this prefix will be added to the keys published to Statsd</li>
59       * </ul>
60       * 
61       * @param properties
62       */
63      @Override
64      public synchronized void setProperties(
65              final Map<String, PoolProperties.InterceptorProperty> properties) {
66          super.setProperties(properties);
67          InterceptorProperty hostnameProp = properties.get("hostname");
68          if (hostnameProp == null) {
69              throw new IllegalArgumentException(
70                      "property \"hostname\" has not been set");
71          }
72          InterceptorProperty portProp = properties.get("port");
73          if (portProp == null) {
74              throw new IllegalArgumentException(
75                      "property \"port\" has not been set");
76          }
77          InterceptorProperty sampleRateProp = properties.get("sampleRate");
78          if (sampleRateProp == null) {
79              throw new IllegalArgumentException(
80                      "property \"sampleRate\" has not been set");
81          }
82          InterceptorProperty prefixProp = properties.get("prefix");
83          if (prefixProp == null) {
84              throw new IllegalArgumentException(
85                      "property \"prefix\" has not been set");
86          }
87  
88          metrics = new Metrics(hostnameProp.getValue(),
89                  portProp.getValueAsInt(0), prefixProp.getValue(),
90                  sampleRateProp.getValueAsDouble(1.0));
91          proxyFactory = new ProxyFactory(metrics);
92      }
93  
94      /**
95       * Gets invoked each time an operation on {@link java.sql.Connection} is
96       * invoked.
97       * 
98       * The timing and count of each method calls to {@link java.sql.Connection}
99       * are sent to Statsd.
100      * 
101      * {@inheritDoc}
102      */
103     @Override
104     public final Object invoke(final Object proxy, final Method method,
105             final Object[] args) throws Throwable {
106         String methodName = method.getName();
107         long start = System.nanoTime();
108         try {
109             Object o = super.invoke(proxy, method, args);
110             if ("createStatement".equals(methodName)) {
111                 o = proxyFactory.statementProxy((Statement) o);
112             } else if ("prepareStatement".equals(method.getName())) {
113                 o = proxyFactory.preparedStatementProxy((PreparedStatement) o);
114             } else if ("prepareCall".equals(method.getName())) {
115                 o = proxyFactory.callableStatementProxy((CallableStatement) o);
116             }
117             return o;
118         } catch (InvocationTargetException e) {
119             throw e.getCause();
120         } finally {
121             metrics.increment(".connection." + methodName + ".count");
122             metrics.timing(".connection." + methodName + ".timing",
123                     System.nanoTime() - start);
124         }
125     }
